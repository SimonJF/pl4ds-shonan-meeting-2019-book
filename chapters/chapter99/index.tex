	\left|\nabla U_{\alpha}^{\mu}(y)\right| &\le&\frac1{d-\alpha}\int
	\left|\nabla\frac1{|\xi-y|^{d-\alpha}}\right|\,d\mu(\xi) =
	\int \frac1{|\xi-y|^{d-\alpha+1}} \,d\mu(\xi)  \\
